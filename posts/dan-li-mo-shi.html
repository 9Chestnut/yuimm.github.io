/**

 *
 * @Author HCT
 * @Date 2021/3/26 20:29
 * @Version 1.0
 */
public class LazyMan {

    private LazyMan(){
    }
     private  static LazyMan lazyMan;
      public static LazyMan getInstance(){
                if (lazyMan == null){
                    lazyMan = new LazyMan();  //不是一个原子性的操作
                    /**
                     * 1.分配内存空间
                     * 2.执行构造方法，初始化对象
                     * 3.把这个对象指向这个空间
                     *
                     * 123
                     *
                     * 因为指令重排可能会执行132的操作
                     * 此时线程A因为执行132的操作顺序，导致对象还没有初始化，但是lazyMan已经不是null了
                     * 当线程B执行的时候，判断lazyMan不是null ，但是此时lazyMan还没有完成构造，出错
                     * 所以需要加 volatile 关键字避免指令重排
                     */
                }
                return lazyMan;
            }   
}

双重检测锁模式

/**
 * 懒汉式单例
 *
 * @Author HCT
 * @Date 2021/3/26 20:29
 * @Version 1.0
 */
public class LazyMan {

    private LazyMan(){
    }

    // 加volatile是因为这是个非原子性的操作，为了避免指令重排
    private volatile static LazyMan lazyMan;

    //双重检测锁模式的 懒汉式单例   SCL懒汉式
    public static LazyMan getInstance(){
        if (lazyMan == null){
            synchronized (LazyMan.class){  //锁住类
                if (lazyMan == null){
                    lazyMan = new LazyMan();  //不是一个原子性的操作
                    /**
                     * 1.分配内存空间
                     * 2.执行构造方法，初始化对象
                     * 3.把这个对象指向这个空间
                     *
                     * 123
                     *
                     * 因为指令重排可能会执行132的操作
                     * 此时线程A因为执行132的操作顺序，导致对象还没有初始化，但是lazyMan已经不是null了
                     * 当线程B执行的时候，判断lazyMan不是null ，但是此时lazyMan还没有完成构造，出错
                     * 所以需要加 volatile 关键字避免指令重排
                     */
                }
            }
        }
        return lazyMan;
    }
}

版本4

两次都使用反射创建对象，避开版本四对构造函数加的锁，再次破坏单例

    //反射破坏单例
    public static void main(String[] args) throws NoSuchMethodException, IllegalAccessException, InvocationTargetException, InstantiationException {
        Constructor<LazyMan> declaredConstructor = LazyMan.class.getDeclaredConstructor(null);
        declaredConstructor.setAccessible(true); //无视私有的构造器
        LazyMan lazyMan2 = declaredConstructor.newInstance();
        LazyMan lazyMan3 = declaredConstructor.newInstance();

        System.out.println(lazyMan3);
        System.out.println(lazyMan2);

    }

解决：设置一个自己定义的boolean值为true，当第一次使用构造函数的时候，设置为false，第二次使用构造的时候条件不允许，不能再定义新的对象

 private static boolean hct = true;
    private LazyMan(){
        synchronized (LazyMan.class){
            if (hct == true){
                hct = false;
            } else {
                throw new RuntimeException("不要试图用反射破坏单例！");
            }
        }
    }

但是代码仍然可能会被反编译，即使是加密过的代码，所以，被反编译后，找到自定义的boolean还是可以通过反射去再次修改为true！道高一尺，魔高一丈！拔剑吧，程序员！
