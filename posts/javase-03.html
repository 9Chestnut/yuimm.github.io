package com.hct.IO;

import java.io.*;

//目录和文件的复制和粘贴
public class Demo6 {
    public static void main(String[] args) throws IOException {

        //拷贝源
        File srcFile = new File("F:\\test1\\2.2");
        //拷贝目标
        File destFile = new File("G:\\");
        //调用方法拷贝
        copyDir(srcFile,destFile);
    }
    private static void copyDir(File srcFile,File destFile){
        if (srcFile.isFile()){
            FileInputStream in = null;
            FileOutputStream out =null;
            try {
                in = new FileInputStream(srcFile);
                String path = ( destFile.getAbsolutePath().endsWith("\\") ? destFile.getAbsolutePath() : destFile.getAbsolutePath() + "\\" ) + srcFile.getAbsolutePath().substring(3);
                out = new FileOutputStream(path);
                //一边读一边写
                byte[] bytes = new byte[1024*1024];
                int readCount = 0;
                while ((readCount = in.read(bytes)) != -1){
                    out.write(bytes,0,readCount);
                }


                out.flush();
            } catch (FileNotFoundException e) {
                e.printStackTrace();
            } catch (IOException e) {
                e.printStackTrace();
            } finally {
                if (in != null){
                    try {
                        in.close();
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
                if (out != null){
                    try {
                        out.close();
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
            }
            return;
        }
        File[] files = srcFile.listFiles();
        for (File file : files){

            if (file.isDirectory()){
                //源目录
                String srcDir = file.getAbsolutePath();
               //目标目录
                String destDir = ( destFile.getAbsolutePath().endsWith("\\") ? destFile.getAbsolutePath() : destFile.getAbsolutePath() + "\\" ) + srcDir.substring(3);
                File newFile = new File(destDir);
                if (!newFile.exists()){
                    newFile.mkdirs();
                }

            }
            //递归调用
            copyDir(file,destFile);
        }
    }
}

序列化和反序列化

​ Java.io.ObjectInputStream

​ Java.io.ObjectOutputStream

  • 序列化：Serialize，将java对象存储到文件中，将java对象的状态保存下来的过程。
  • 反序列化：DeSerialize，将硬盘上的数据重新恢复到内存当中，恢复成java对象。
  • 参与的对象必须要实现Serializable这个接口，这个接口只是一个标志性接口，里面什么也没有，虚拟机看到这个接口之后，会默认提供序列化版本号
  • 一个不参与序列化的属性，只需要给属性加上transient。
  • 自动化生成序列化版本号的缺点：一旦代码确定以后，就不能就行后续的修改。因为只要修改，必然会重新编译，此时会生成全新的序列化版本号，这个时候java虚拟机会认为这是一个全新的类。建议：凡是一个类实现了Serializable接口，建议给该类一个固定不变的版本号（手动写出来），这样以后即使代码修改了，这个版本号不变，java虚拟机会认为这是同一个类。

生产者和消费者

package com.hct.threadsafe;

import java.util.ArrayList;
import java.util.List;

public class Demo {
    public static void main(String[] args) {
        //创建一个仓库对象，共享的
        List list = new ArrayList();
        //创建两个线程对象
        Thread t1 = new Thread(new Producer(list));
        Thread t2 = new Thread(new Consumer(list));
        t1.setName("生产者线程");
        t2.setName("消费者线程");
        t1.start();
        t2.start();



    }
}


//生产者线程
class Producer implements Runnable {

    //仓库
    private List list;

    public Producer(List list){
        this.list = list;
    }


    @Override
    public void run() {
        //一直生产
        while (true){
            synchronized (list){
                if (list.size() > 0){
                    //仓库满了
                    //当前线程进入等待状态,并且释放list锁
                    try {
                        list.wait();
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
                Object obj = new Object();
                list.add(obj);
                System.out.println(Thread.currentThread().getName() + "--->" + obj);
                //唤醒消费者消费
                list.notify();
            }
        }

    }
}

//消费者线程
class Consumer implements Runnable {

    //仓库
    private List list;

    public Consumer(List list){
        this.list = list;
    }

    @Override
    public void run() {
        //一直消费
        while (true){
            synchronized (list){
                if (list.size() == 0){
                    //仓库空了
                    //当前线程进入等待状态,并且释放list锁
                    try {
                        list.wait();
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
                Object obj = list.remove(0);
                System.out.println(Thread.currentThread().getName() + "---->" + obj);
                //唤醒生产者
                list.notify();
            }
        }
    }
}

反编译构造方法

package com.hct.reflext;

import java.lang.reflect.Constructor;
import java.lang.reflect.Modifier;

public class Demo4 {
    public static void main(String[] args) {

        try {
            StringBuilder s = new StringBuilder();
            Class user = Class.forName("com.hct.bean.User");
            s.append(Modifier.toString(user.getModifiers()));
            s.append(" class ");
            s.append(user.getSimpleName());
            s.append("{\n");

            //拼接构造方法
            Constructor[] constructors = user.getDeclaredConstructors();
            for (Constructor constructor : constructors) {
                s.append("\t");
                s.append(Modifier.toString(constructor.getModifiers()));
                s.append(" ");
                s.append(user.getSimpleName());
                s.append("(");
                //拼接参数
                Class[] parameterTypes = constructor.getParameterTypes();
                for (Class parameterType : parameterTypes) {
                    s.append(parameterType.getSimpleName());
                    s.append(",");

                } 
                //删除最后下标位置上的字符
                if (parameterTypes.length > 0){
                    s.deleteCharAt(s.length() - 1);
                }
                s.append("){}\n");
            }

            s.append("}");
            System.out.println(s);

        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        }

    }
}

  1. 使用格式： @注解类型名

  2. 注解可以出现在类上，属性上，方法上，变量上，参数列表上等，还可以出现在注解类型上。

  3. 用来标注注解的注解叫做元注解。常见的元注解：

    Target ：表示“被标注的注解”可以出现在哪些位置上

    Retention：表示“被标注的注解”最终保存在哪里

    Deprecated：表示注解的元素已经过时

反射注解

注解：

package com.hct.annotation1;


import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

@Target({ElementType.TYPE,ElementType.METHOD})
@Retention(RetentionPolicy.RUNTIME)
public @interface MyAnnotation1 {

    String value() default "北京";

}

使用注解的类：

package com.hct.annotation1;

@MyAnnotation1("shamghai")
public class MyAnnotationTest {
    int i;
    public MyAnnotationTest(){
    }
    @MyAnnotation1("qwewerr")
    public void doSomr(){

    }
}

测试类：

package com.hct.annotation1;

public class Demo1 {
    public static void main(String[] args) throws ClassNotFoundException {
        //获取这个类
        Class c = Class.forName("com.hct.annotation1.MyAnnotationTest");

        //获取类上的注解
        if (c.isAnnotationPresent(MyAnnotation1.class)){
            //获取注解对象
            MyAnnotation1 myAnnotation1 =  (MyAnnotation1)c.getAnnotation(MyAnnotation1.class);
            //获取注解对象的属性，和调接口没区别
            String value = myAnnotation1.value();
            System.out.println(value);
        }

        //获取方法上的注解
        Method method = c.getDeclaredMethod("doSome");
        if (method.isAnnotationPresent(MyAnnotation1.class)){
            //获取注解对象
            MyAnnotation1 myAnnotation1 =  (MyAnnotation1)method.getAnnotation(MyAnnotation1.class);
            //获取注解对象的属性
            System.out.println(myAnnotation1.value());
        }

    }
}
